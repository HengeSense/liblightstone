PROJECT(LIBLIGHTSTONE)
SET(EXECUTABLE_OUTPUT_PATH ${CMAKE_CURRENT_SOURCE_DIR}/bin)
SET(LIBRARY_OUTPUT_PATH ${CMAKE_CURRENT_SOURCE_DIR}/lib)
SET(LIBLSVERSION 0.1)
IF(WIN32)
  SET(LIBLIGHTSTONENAME "liblightstone")
  SET(LIBHIDNAME "libhid")
  SET(LIBUSBNAME "libusb")
ELSE(WIN32)
  SET(LIBLIGHTSTONENAME "lightstone")
  SET(LIBHIDNAME "hid")
  SET(LIBUSBNAME "usb")
  INCLUDE_DIRECTORIES (/usr/include /usr/local/include)
  LINK_DIRECTORIES(/usr/lib /usr/local/lib)
  IF (APPLE)
    SET(OTHER_LDFLAGS " -framework Carbon -framework IOKit")
  ENDIF (APPLE)
ENDIF(WIN32)
ADD_SUBDIRECTORY(src)
ADD_SUBDIRECTORY(examples/lightstone_test)
